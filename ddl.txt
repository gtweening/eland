Create table TblMaterials (
Id int not null auto_increment primary key , 
Omschr varchar(50)
);

Create table TblCheckpoints (
Id int not null auto_increment primary key , 
Omschr varchar(50)
);

Create table TblSections (
Id int not null auto_increment primary key , 
Naam varchar(5) not null,
Omschr varchar(30)
);

Create table TblObstacles (
Id int not null auto_increment primary key,
Section_id INT NOT NULL REFERENCES TblSections(Id),
Volgnr int not null,
Omschr varchar(50),
ImgPath varchar(120),
ChkQ1 boolean,
ChkQ2 boolean,
ChkQ3 boolean,
ChkQ4 boolean
);

Create table TblObstacleMaterials (
Id int not null auto_increment primary key,
Obstacle_id INT NOT NULL REFERENCES TblObstacles(Id),
Material_id INT NOT NULL REFERENCES TblMaterials(Id),
Aantal varchar(15)
);

Create table TblObstacleCheckpoints (
Id int not null auto_increment primary key,
Obstacle_id INT NOT NULL REFERENCES TblObstacles(Id),
Checkpoint_id INT NOT NULL REFERENCES TblCheckpoints(Id)
);

create table TblObstacleChecks (
Id int not null auto_increment primary key , 
Obstacle_id INT NOT NULL REFERENCES TblObstacles(Id),
DatCheck date,
ChkSt boolean,
CheckedBy varchar(30),
Note TEXT CHARACTER SET utf8
);

Create table TblUsers (
Id int not null auto_increment primary key , 
Email varchar(100) not null,
Password varchar(128) not null,
salt char(128) not null,
Admin boolean
);

CREATE TABLE login_attempts (
    user_id int NOT NULL,
    time VARCHAR(30) NOT NULL
)
